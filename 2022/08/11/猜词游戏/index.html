猜词游戏

发表于2022-08-11|更新于2022-08-11|python
|字数总计:2k|阅读时长:8分钟|阅读量:

在家没事，写了个小游戏

"""
猜数词游戏-1.0
author: HomesIII
字典目录[dictionary_path]
字典格式:
    word explain(中文等双字节字符)
    word explain(中文等双字节字符)
    ...  ...
"""
import re
from time import sleep
from sys import exit as s_exit
from random import randint

errors = 0
# 用字典目录替换这个
dictionary_path = '[dictionary_path]'
def choose_word():
    wd = ''
    global errors # 统计错误，防止递归过深
    line_counter = 0 #行数
    counter = 0 #目标词所在行，用random.randint生成
    try:
        with open(dictionary_path, 'r') as f:
            for _ in f:
                line_counter += 1
            line = randint(1, line_counter)
        with open(dictionary_path, 'r') as file:
            for w in file:
                counter += 1
                if counter == line:
                    wd = w.strip()
    except FileNotFoundError:
        print('无法打开指定的文件!')
    except LookupError:
        print('指定了未知的编码!')
    except UnicodeDecodeError:
        print('读取文件时解码错误!')
    except IOError:
        print('读写文件时出现错误.')
    try:
        words = re.search(r'(\w*\s?\w*)(?=\s)', wd).group(0)
        explains = re.search(r'(?<=\s)[^\x00-\xff]+', wd).group(0)
        # 判断wd是否为空，若为空则递归调用choose_word
        # 若字典格式错误，words, explains 为空也会报错
        return words, explains
    except AttributeError:
        errors += 1
        if errors == 10:
            print('文件错误过多, 已停止递归')
            s_exit()
        print(f'第{line}行: :{wd}: 格式错误')
        print("正确格式: 单词 解释")
        return choose_word()
print("这是一个回合制猜词游戏.按.h显示帮助")
i = input("回车开始")
round_ = 0 #回合数
tip_rule = 0 # 提示频率(猜几次提示一次)
auto_play = False
sleep_time = 1
while not i:
    word, explain = choose_word()#选词
    round_ += 1
    print()
    print(f'-------第{round_}回合--------')
    display_num = 1 # 展示的字母数量
    time = 0 # 猜的次数, 可以设置提示频率
    while not (ipt := input(f"这个单词以{word[:display_num]}开头: ")) or ipt[:2] != '.q':
        time += 1
        if ipt == '.h':
            time -= 1 #每次使用特殊命令, 猜的次数减一(特殊命令不计入猜测次数)
            print()
            print('------帮助-------')
            print('.h   ---帮助')
            print('.q   ---退出')
            print('.t   ---提示')
            print('.l   ---提示单词长度')
            print('.g   ---放弃')
            print('.s   ---设置')
            print('         |_学习模式自动进入下一个')
            print('         |_自动学习模式间隔时间')
            print('         |_自动提示频率')
            print()
            print('.k   ---学习')
            print()
        elif ipt == word:
            print('猜对啦')
            print(word + ', '+explain)
            break
        elif ipt == '.t':
            time -= 1
            display_num += 1
            if display_num == len(word):
                print('这个单词是' + word + '!')
                print('解析: ' + explain)
                break
            continue
        elif ipt == '.l':
            time -= 1
            print(f'这个单词有{len(word)}个字母')
        elif ipt == '.s':
            time -= 1
            print()
            print(f'1 ---学习模式自动进入下一个({str(auto_play)})')
            print(f'2 ---自动学习模式间隔时间({sleep_time}秒)')
            print(f'3 ---提示频率({tip_rule})')
            print()
            choose = input('选择设置项(1/2/3): ')
            if choose == '1':
                auto_play = not auto_play # 旧版实现: input('学习模式自动进入下一个[(y)/n]') != 'n'
                print('设置成功, 间隔时间:', sleep_time, '秒')
            elif choose == '2':
                conti = True
                while conti:
                    try:
                        user_sleep = input('自动学习模式间隔时间(秒): ')
                        if not user_sleep:
                            print('已取消')
                            break
                        sleep_time = float(user_sleep)
                        conti = False
                        print('设置成功, 间隔时间:', sleep_time, '秒')
                    except Exception:
                        print('请输入数字')
            elif choose == '3':
                tip = tip_rule
                while tip == tip_rule:
                    rule = input('设置提示频率(默认不自动提示): ') if tip_rule == 0 else input(f'设置提示频率(猜{tip_rule}次提示一次): ')
                    if not rule:
                        print("已取消")
                        break
                    try:
                        int(rule)
                    except Exception:
                        print('请输入数字, 回车取消')
                        continue
                    tip_rule = int(rule)
                    print("设置成功")
                    break # break是为了避免用户重复设置同一值的时候进入死循环一直提示输入(虽然回车可以取消)
            else:
                print('取消.')
        elif ipt == '.g':
            print(f'加油⛽, 这个单词是{word}, ' + explain)
            break
        elif ipt == '.k':
            print('-----学习模式-----')
            input('回车键继续')
            try:
                with open(dictionary_path, 'r') as reader:
                    for lines in reader:
                        print(lines, end = '')
                        if auto_play:
                            sleep(sleep_time)
                            continue
                        input()
                    print()
                        
            except Exception as e:
                print('文件读写错误')
                print('异常信息: \n', e)
        else:
            print('猜错了')
        if tip_rule != 0 and time != 0 and time % tip_rule == 0:
            print()
            display_num += 1
            if display_num == len(word):
                print('这个单词是' + word + '!\n解析: explain')
                break # 提示单词加一后立即判断是否猜出, 若未猜出，再输出提示
            print(f"提示:\n这个单词以{word[:display_num]}开头")
    else:
        s_exit() if input('退出?[(y)/n]') in 'Yy\n' else None
else:
    print("退出......")
    s_exit()
