可以看见，每一张的标题和链接都被<li>包裹
href=是文章的相对路径,
紧跟着的是文章标题.
我们按照先抓大在再抓小的原则,
先整体抓取<ol>中的部分:

import requests
import re #使用正则表达式

url = 'https://gigamonkeys.com/book/'

html = requests.get(url).text
urls = re.search("<ol.*?</ol>", html).group(1)

然后抓取名称和链接:

chapter_links = re.findall("href='(.*?)'", urls, re.S)
chapter_names = re.findall("'>(.*?)</a", urls, re.S)

搞完以后，把它们下载下来

from os import makedirs
import requests

makedirs("commonlisp", exist_ok = True)
for name, link in zip(chapter_names, chapter_links):
    print("downloading " + name + "...")
    with open("commonlisp/" + link, "w") as file:
        file.write(requests.get(url + link).text)
print("done.")

这个爬虫程序可以说是非常简单，因为它只有源代码爬取功能，没有做数据清洗，所以不是给人看的，而且因为是单线程，速度方面不够优秀，不过范例章节比较少，也算是差强人意

完整代码:

try:
    from os import makedirs, path
    import re

    import requests
except Exception as e:
    print("导入包出错:",  e, "\n请检查环境是否完整")
    exit()
url = 'https://gigamonkeys.com/book/'
try:
    html = requests.get(url).text
except Exception as e:
    print("网络错误, 获取网页失败")
    exit()

urls = re.search("<ol(.*?)</ol>", html, re.S).group(1)
chapter_name = re.findall("'>(.*?)</a", urls, re.S)
chapter_link = re.findall("href='(.*?)'", urls, re.S)

makedirs("commonlisp", exist_ok = True)
for name, link in zip(chapter_name, chapter_link):
    print("Downloading", name, ". . . ", end = '')
    try:
        with open(path.join('commonlisp', link), 'w') as file:
            file.write(requests.get(url + link).text)
            print("done")
    except Exception as e:
        print("error")
        print(e)
print("complete")

哈哈，被try except淹没的感觉

线程池优化版本
try..except: 战战兢兢，如临深渊，如履薄冰 (•̥́_•ૅू˳)

import re
import requests
from os import makedirs
from multiprocessing.dummy import Pool

def save(name, link):
    if name and link:
        exit()
    makedirs("commonlisp", exist_ok=True)
    with open("commonlisp" + link, 'w') as f:
        f.write(requests.get(url + link).text)
        print(name)
        
url = 'https://gigamonkeys.com/book/'
html = requests.get('https://gigamonkeys.com/book/').text

urls = re.search("<ol(.*?)</ol>", html, re.S).group(1)
chapter_name = re.findall("'>(.*?)</a", urls, re.S)
chapter_link = re.findall("href='(.*?)'", urls, re.S)

pool = Pool(4)
#with Pool(4) as pool:
    #pool.starmap(save, zip(chapter_name, chapter_link))
pool.starmap(save, zip(chapter_name, chapter_link))
pool.join()
pool.close()
